# HRIS Vacancy
CREATE TABLE "HRIS_VACANCY" 
   (    "VACANCY_ID" NUMBER(7,0) NOT NULL ENABLE, 
    "POSITION" VARCHAR2(200 BYTE) NOT NULL ENABLE, 
    "AVAILABILITY" NUMBER(7,0) NOT NULL ENABLE, 
    "DESCRIPTION" VARCHAR2(4000 BYTE) NOT NULL ENABLE, 
    "STATUS" CHAR(1 BYTE) DEFAULT 1, 
    "CREATED_DT" DATE, 
    "MODIFIED_DT" DATE, 
    "CREATED_BY" NUMBER(7,0), 
    "MODIFIED_BY" NUMBER(7,0), 
    "VACANCY_STATUS" NUMBER DEFAULT 1, 
     CONSTRAINT "PK_VACANCY_ID" PRIMARY KEY ("VACANCY_ID")
  
   ) 

# HRIS_VACANCY_APPLICATION
 CREATE TABLE "HRIS_VACANCY_APPLICATION" 
   (    "ID" NUMBER(7,0) NOT NULL ENABLE, 
    "POSITION" VARCHAR2(255 BYTE), 
    "FULL_NAME" VARCHAR2(255 BYTE), 
    "GENDER" CHAR(1 BYTE), 
    "EMAIL" VARCHAR2(255 BYTE), 
    "CONTACT_NUMBER" NUMBER(10,0), 
    "COVER_LETTER" VARCHAR2(2000 BYTE), 
    "RESUME_PATH" VARCHAR2(255 BYTE), 
    "STATUS" CHAR(2 BYTE), 
    "APPLICANT_QUESTION" VARCHAR2(256 BYTE), 
    "FEFRENCE_FROM" VARCHAR2(256 BYTE), 
     CONSTRAINT "PK_HRIS_VACANCY_APPLICATION_ID" PRIMARY KEY ("ID"))

# Create view REQUIRED_RESUME_DETAILS
CREATE VIEW REQUIRED_RESUME_DETAILS as
select a.FULL_NAME, a.POSITION, b.DESCRIPTION, a.RESUME_PATH from HRIS_VACANCY_APPLICATION a JOIN HRIS_VACANCY b
on a.ID = b.VACANCY_ID
